# 箱型图

比较不同数据的平均数和p-value值

```R
library(ggplot2)
library(ggpubr)

leaf=read.table("C:\\Users\\PC\\Desktop\\R语言作图脚本\\R作图文件\\leaf.plot")
colnames(leaf)=c("layer","fpkm","days")
leaf$layer=factor(leaf$layer,levels=c("A2_>0DHS","A2_0DHS","D5_>0DHS","D5_0DHS","AD1_>0DHS","AD1_0DHS"))
leaf$days=factor(leaf$days,levels = c("A2","D5","AD1"))
ggplot(leaf,aes(x=layer,y=fpkm,fill=days))+
geom_boxplot(aes(group=layer),outlier.color=NA)+
scale_fill_manual(values = c("#FFCCCC","#99CCCC","#336699"))+
theme(panel.grid = element_blank())+
theme(axis.line = element_line(size=1,colour = "black"),axis.text.x = element_text(angle = 45,hjust=0.5,vjust=0.5,colour="black"))+
theme(panel.background=element_blank())+
stat_compare_means(comparisons = list(c("A2_>0DHS","A2_0DHS"),c("D5_>0DHS","D5_0DHS"),c("AD1_>0DHS","AD1_0DHS")),label="p.signif",method="wilcox.test",label.y=10)+
theme(plot.title = element_text(hjust = 0.5))


```
